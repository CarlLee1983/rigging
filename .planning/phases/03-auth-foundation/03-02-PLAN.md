import { type Brand, brand } from '../../../shared/kernel/brand'
import { ValidationError } from '../../../shared/kernel/errors'

export type Email = Brand<string, 'Email'>

// Minimal validator — not full RFC 5322. BetterAuth does its own email parsing; this guards
// our domain boundary against empty / whitespace / obviously-invalid input before we hand it over.
const EMAIL_RE = /^[^\s@]+@[^\s@]+\.[^\s@]+$/

export const Email = (value: string): Email => {
  const normalized = value.trim().toLowerCase()
  if (!EMAIL_RE.test(normalized)) {
    throw new ValidationError(`Invalid email: ${JSON.stringify(value)}`)
  }
  return brand<'Email'>()(normalized)
}
```

 * getApiKeyService — AUX-05 Runtime Guard factory.
 *
 * The ONLY legal entry point to the internal ApiKeyService. Any caller that attempts to
 * `import { ApiKeyService } from '.../internal/api-key-service'` directly is blocked by Biome
 * (application/presentation noRestrictedImports.patterns).
 *
 * Factory throws AuthContextMissingError with a 4-section teaching message if ctx is missing
 * (null / undefined / falsy userId). Message format per Pitfall #9 + P1 D-12:
 *   1. What happened (one-liner)
 *   2. Reason:
 *   3. See docs/decisions/0006-authcontext-boundary.md
 *   4. Fix: code example
 *
 * Don't shorten the message — agents read it; one-liners are useless for repair.
 */
export const getApiKeyService = (ctx: AuthContext | null | undefined): ApiKeyService => {
  if (!ctx?.userId) {
    throw new AuthContextMissingError(
      `AuthContext is missing when calling getApiKeyService(ctx).

Reason: Domain services require AuthContext from \`requireAuth: true\` macro. This is Rigging Rigidity Map Tier 1 rule 1 (AGENTS.md) — no route should reach a domain factory without having declared its auth posture.
See docs/decisions/0006-authcontext-boundary.md and docs/decisions/0007-runtime-guards-via-di.md.

Fix: Declare \`requireAuth: true\` in your route options. Example:
  new Elysia()
    .use(authContextPlugin(identity))
    .get('/api-keys', ({ authContext }) => {
      const svc = getApiKeyService(authContext)
      return svc.ctx
    }, { requireAuth: true })`
    )
  }
  return new ApiKeyService(ctx)
}

// Port (interface) — application layer defines what it needs; infrastructure implements.
// Rigging wraps BetterAuth via this port; any future auth implementation (OAuth, passkey, v2)
// swaps the adapter while keeping use cases unchanged.
//
// Timing-align contract (D-10): verifyApiKey MUST run uniform-latency code paths — malformed key
// and valid-format-but-wrong-hash both exercise dummy timingSafeEqual + dummy DB SELECT. See
// .planning/phases/03-auth-foundation/03-RESEARCH.md Pattern 3 L473-510 for the adapter shape.

import type { AuthContext, Scope, UserId } from '../../domain/auth-context'

export interface IIdentityService {
  /**
   * Parse cookies from `headers` and resolve to AuthContext if a valid session exists.
   * D-03 mapping: BetterAuth session.user.id → AuthContext.userId (branded via `as UserId`),
   *               identityKind='human', scopes=['*'], sessionId=session.id.
   * Returns null if headers carry no valid session. Never throws for auth-failure reasons.
   */
  verifySession(headers: Headers): Promise<AuthContext | null>

  /**
   * Resolve raw `x-api-key` header to AuthContext.
   * D-10 timing alignment: implementation MUST execute dummy timingSafeEqual + dummy DB SELECT on
   *                        the malformed-path + wrong-hash-path so latency does not leak reject cause.
   * D-11 shape: returns { userId, identityKind:'agent', scopes, apiKeyId } — NO sessionId.
   * Returns null on any validation failure (malformed / unknown prefix / wrong hash / revoked / expired).
   */
  verifyApiKey(rawKey: string): Promise<AuthContext | null>

  /**
   * Create an API key for `params.userId`. Called by CreateApiKeyUseCase AFTER it enforces
   * AUTH-15 (userId === ctx.userId) and D-04 (scopes ⊆ session.scopes).
   * Returns { id, rawKey, prefix, createdAt }. The rawKey is visible ONLY at this return point —
   * D-20: controllers surface it to the client once; DB stores hash.
   */
  createApiKey(params: {
    userId: UserId
    label: string
    scopes: ReadonlyArray<Scope>
    expiresAt: Date
  }): Promise<{ id: string; rawKey: string; prefix: string; createdAt: Date }>

  /**
   * List all non-revoked + non-expired keys belonging to `userId`.
   * Returned shape intentionally does NOT include the hash — API consumers see metadata only.
   */
  listApiKeysByUser(userId: UserId): Promise<
    Array<{
      id: string
      label: string
      prefix: string
      scopes: ReadonlyArray<string>
      expiresAt: Date | null
      createdAt: Date
      revokedAt: Date | null
    }>
  >

  /**
   * Soft-delete (D-24): sets `revokedAt = now()` iff the key exists AND belongs to `userId`.
   * No-op if the key doesn't exist or belongs to another user — the use case surfaces
   * NotFoundError for the mismatch case (controller decides HTTP mapping).
   */
  revokeApiKey(id: string, userId: UserId): Promise<void>
}
```
