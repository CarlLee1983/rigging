import { createHash, timingSafeEqual } from 'node:crypto'
import type { AuthContext, UserId } from '../../domain'
import type { IApiKeyRepository } from '../../application/ports/api-key-repository.port'
import type { IIdentityService } from '../../application/ports/identity-service.port'
import type { AuthInstance } from './auth-instance'

// D-19 canonical prefix format: 'rig_live_' + 43 chars (32 bytes base64url).
// Total raw key length ≈ 9 + 43 = 52 chars.
const API_KEY_PREFIX = 'rig_live_'
const PREFIX_INDEX_LEN = 8         // D-21: first 8 chars of the raw key go into the indexed prefix column
const MIN_RAW_KEY_LEN = API_KEY_PREFIX.length + 20  // lower-bound sanity check before heavy work

// Pre-computed dummy 32-byte buffer for timing-align malformed path (D-10).
// We must NEVER pass mismatched-length buffers to timingSafeEqual (Pitfall #3 throws RangeError).
const DUMMY_HASH = createHash('sha256').update('rigging-dummy').digest()

/**
 * BetterAuthIdentityService — implements IIdentityService by wrapping BetterAuth's server API
 * plus a direct IApiKeyRepository lookup for the resolver path (D-10 timing alignment).
 *
 * Why two data paths? BetterAuth's apiKey plugin has its own verify flow but doesn't expose
 * timing-uniform failure paths. Rigging hits IApiKeyRepository directly for verifyApiKey to
 * guarantee D-10: malformed-format and valid-format-wrong-hash paths execute the same DB SELECT +
 * timingSafeEqual operations so latency cannot leak the rejection cause.
 */
export class BetterAuthIdentityService implements IIdentityService {
  constructor(
    private readonly auth: AuthInstance,
    private readonly apiKeys: IApiKeyRepository,
  ) {}

  async verifySession(headers: Headers): Promise<AuthContext | null> {
    const result = await this.auth.api.getSession({ headers })
    if (!result) return null
    return {
      userId: result.user.id as UserId,
      identityKind: 'human',            // D-03
      scopes: ['*'],                    // D-03 human defaults to sudo
      sessionId: result.session.id,
    }
  }

  async verifyApiKey(rawKey: string): Promise<AuthContext | null> {
    // D-10 STEP 1 — Format check. Malformed path runs the same shape of work as the real path
    // (one timingSafeEqual + one findByPrefix) so latency is observationally indistinguishable.
    if (!rawKey.startsWith(API_KEY_PREFIX) || rawKey.length < MIN_RAW_KEY_LEN) {
      timingSafeEqual(DUMMY_HASH, DUMMY_HASH)                    // dummy same-length compare
      await this.apiKeys.findByPrefix('xxxxxxxx')                // dummy DB SELECT (no row matches)
      return null
    }

    // D-21 STEP 2 — Prefix lookup via indexed column.
    const prefix = rawKey.slice(0, PREFIX_INDEX_LEN)
    const row = await this.apiKeys.findByPrefix(prefix)

    // D-24 soft-delete + D-22 expiry — null / revoked / expired all reject uniformly.
    if (!row || row.revokedAt !== null || (row.expiresAt && row.expiresAt.getTime() <= Date.now())) {
      timingSafeEqual(DUMMY_HASH, DUMMY_HASH)                    // dummy compare so no-row path also burns one compare
      return null
    }

    // D-23 STEP 3 — Hash compare.
    const computedHash = createHash('sha256').update(rawKey).digest()
    const storedHashBuf = Buffer.from(row.hash, 'hex')

    // Pitfall #3 — timingSafeEqual throws on length mismatch. Pre-check, fall through to dummy.
    if (computedHash.length !== storedHashBuf.length) {
      timingSafeEqual(DUMMY_HASH, DUMMY_HASH)
      return null
    }
    if (!timingSafeEqual(computedHash, storedHashBuf)) {
      return null
    }

    // D-11 — agent path returns NO sessionId.
    return {
      userId: row.userId as UserId,
      identityKind: 'agent',
      scopes: row.scopes,
      apiKeyId: row.id,
    }
  }

  async createApiKey(params: {
    userId: UserId; label: string; scopes: ReadonlyArray<string>; expiresAt: Date
  }): Promise<{ id: string; rawKey: string; prefix: string; createdAt: Date }> {
    // Delegate to BetterAuth's apiKey plugin. BetterAuth 1.6.5 body shape (confirmed via Context7
    // in RESEARCH.md — apiKey plugin reference):
    //   body: { userId, name, prefix?, expiresIn?, scopes?, metadata? }
    // We pass our own prefix per D-19 so the CLI-generated `prefix` column gets the 'rig_live_' format.
    // `expiresIn` is milliseconds-from-now per BetterAuth contract; compute from input.expiresAt.
    const expiresInMs = params.expiresAt.getTime() - Date.now()
    const created = await this.auth.api.createApiKey({
      body: {
        userId: params.userId as string,
        name: params.label,
        prefix: API_KEY_PREFIX,
        expiresIn: expiresInMs,
        // D-07/Q2 — store scopes as metadata (BetterAuth's built-in `permissions` has object shape;
        // we keep the flat string[] shape via metadata to match D-01 ALLOWED_SCOPES vocabulary).
        metadata: { scopes: params.scopes },
      },
    })
    // BetterAuth returns { key, apiKey } — names may vary slightly in 1.6.5; consult the generated
    // schema + node_modules type defs. Adjust property access below if the actual return differs.
    return {
      id: created.apiKey?.id ?? (created as { id: string }).id,
      rawKey: created.key ?? (created as { key: string }).key,
      prefix: created.apiKey?.prefix ?? API_KEY_PREFIX,
      createdAt: created.apiKey?.createdAt ?? new Date(),
    }
  }

  async listApiKeysByUser(userId: UserId) {
    const rows = await this.apiKeys.listByUserId(userId)
    return rows.map((r) => ({
      id: r.id,
      label: r.label,
      prefix: r.prefix,
      scopes: r.scopes,
      expiresAt: r.expiresAt,
      createdAt: r.createdAt,
      revokedAt: r.revokedAt,
    }))
  }

  async revokeApiKey(id: string, userId: UserId): Promise<void> {
    await this.apiKeys.markRevoked(id, userId)
  }
}

    // the sentinel prefix — we do NOT short-circuit to return null without querying.
    const rows = await this.db
      .select()
      .from(apiKey)
      .where(
        and(
          eq(apiKey.prefix, prefix),
          // BetterAuth enabled flag — adjust if schema uses revokedAt column instead.
          eq(apiKey.enabled, true),
          or(isNull(apiKey.expiresAt), gt(apiKey.expiresAt, sql`now()`)),
        ),
      )
      .limit(1)
    if (rows.length === 0) return null
    return ApiKeyMapper.toDomain(rows[0] as never)
  }

  async listByUserId(userId: UserId): Promise<ApiKeyRow[]> {
    const rows = await this.db.select().from(apiKey).where(eq(apiKey.userId, userId as string))
    return rows.map((r) => ApiKeyMapper.toDomain(r as never))
  }

  async markRevoked(id: string, userId: UserId): Promise<boolean> {
    const result = await this.db
      .update(apiKey)
      .set({ enabled: false, updatedAt: new Date() })   // or `.set({ revokedAt: new Date() })` if column exists
      .where(and(eq(apiKey.id, id), eq(apiKey.userId, userId as string)))
      .returning({ id: apiKey.id })
    return result.length > 0
  }
}

import { ScopeNotSubsetError, UserIdMismatchError } from '../../domain/errors'
import type { AuthContext, Scope, UserId } from '../../domain'
import type { IIdentityService } from '../ports/identity-service.port'

export interface IClock {
  now(): Date
}

export interface CreateApiKeyInput {
  userId?: string              // optional in body — CVE vector; if present MUST equal ctx.userId
  label: string
  scopes: ReadonlyArray<Scope>
  expiresAt?: Date
}

export interface CreatedApiKeyDto {
  id: string
  key: string                  // D-20 raw key returned ONCE; never leaked again
  prefix: string
  label: string
  scopes: ReadonlyArray<Scope>
  expiresAt: Date
  createdAt: Date
}

/**
 * CreateApiKeyUseCase — the Rigging value-add for AUTH-12 / AUTH-15 / AUTH-16.
 *
 * Line 1: AUTH-15 + CVE-2025-61928 defense — body.userId MUST match ctx.userId.
 * Line 2: D-04 subset check — requested key scopes ⊆ session scopes.
 * Line 3: Delegate to BetterAuth via IIdentityService.createApiKey (D-22 default 90 days).
 *
 * These three checks are defense-in-depth on top of BetterAuth's own session handling — Rigging
 * does NOT trust "BetterAuth validates session so one check is enough" (Pitfall #5).
 */
export class CreateApiKeyUseCase {
  constructor(
    private readonly identity: IIdentityService,
    private readonly clock: IClock,
  ) {}

  async execute(ctx: AuthContext, input: CreateApiKeyInput): Promise<CreatedApiKeyDto> {
    // === LINE 1: AUTH-15 + CVE-2025-61928 defense ===
    if (input.userId && input.userId !== ctx.userId) {
      throw new UserIdMismatchError('Cannot create API Key for another user')
    }

    // === LINE 2: D-04 subset check ===
    // '*' is a catch-all per D-07 — if session has '*', any requested scope passes subset.
    if (!input.scopes.every((s) => ctx.scopes.includes(s) || ctx.scopes.includes('*'))) {
      throw new ScopeNotSubsetError('Requested key scopes must be subset of your session scopes')
    }

    // === LINE 3: Compute expiry + delegate ===
    const now = this.clock.now()
    const ninetyDaysMs = 90 * 24 * 60 * 60 * 1000
    const expiresAt = input.expiresAt ?? new Date(now.getTime() + ninetyDaysMs)   // D-22
    const label = input.label.trim()                                              // D-25

    const created = await this.identity.createApiKey({
      userId: ctx.userId,                // NOT input.userId — explicit narrowing after guard
      label,
      scopes: input.scopes,
      expiresAt,
    })

    return {
      id: created.id,
      key: created.rawKey,               // D-20 raw key leaves the backend exactly here
      prefix: created.prefix,
      label,
      scopes: input.scopes,
      expiresAt,
      createdAt: created.createdAt,
    }
  }
}
