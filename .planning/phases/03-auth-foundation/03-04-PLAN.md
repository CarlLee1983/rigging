```typescript
import { Elysia, status } from 'elysia'
import type { AuthContext } from '../../domain/auth-context'
import type { IIdentityService } from '../../application/ports/identity-service.port'

/**
 * authContextPlugin — AUX-02 single-root macro mount, scope: global.
 *
 * Exposes one opt-in marker: `requireAuth: true`. Routes marked with it get `ctx.authContext: AuthContext`
 * via the resolver pipeline. Routes NOT marked cannot access `ctx.authContext` (TS error).
 *
 * Resolver precedence (D-09 / D-10 / D-11):
 *   1. If `x-api-key` header present — MUST validate or 401. NO fallback to cookie.
 *   2. If header absent — try cookie session. None → 401.
 *
 * All 401 scenarios return the same generic body via the global error handler
 * (src/shared/presentation/plugins/error-handler.plugin.ts reads UnauthenticatedError.httpStatus + code).
 * Server-side `log.warn` distinguishes causes — response body does not (D-12 / Pitfall #13).
 */
export function authContextPlugin(identity: IIdentityService) {
  return new Elysia({ name: 'rigging/auth-context' })
    .macro({
      requireAuth: {
        async resolve({ request: { headers } }) {
          // D-09/D-10: x-api-key path is first and does NOT fall back on failure.
          const rawApiKey = headers.get('x-api-key')
          if (rawApiKey !== null) {
            const ctx = await identity.verifyApiKey(rawApiKey)
            if (!ctx) return status(401, { error: { code: 'UNAUTHENTICATED', message: 'Authentication required' } })
            return { authContext: ctx satisfies AuthContext }
          }

          // Cookie session path — D-03 human defaults.
          const sessionCtx = await identity.verifySession(headers)
          if (!sessionCtx) return status(401, { error: { code: 'UNAUTHENTICATED', message: 'Authentication required' } })
          return { authContext: sessionCtx satisfies AuthContext }
        },
      },
    })
}
```

**Note on 401 body:** The `status(401, body)` return in the macro produces a RESPONSE directly — it bypasses the `onError` handler. This means the `requestId` field in the body will NOT be present (requestLogger.derive sets `set.headers['x-request-id']` but the body we construct here doesn't include it). For uniform body shape per D-12, we have two choices:

  **Option A (simpler):** Throw `UnauthenticatedError` instead of `status(401, ...)`. The global error handler adds `requestId` via its uniform body shape.

  **Option B (explicit):** Read `requestId` from the ctx parameter of resolve and include it in the body.

**Executor picks Option A** — throwing `UnauthenticatedError` from the resolver. BetterAuth + Elysia macro docs show `resolve` may throw; the global `errorHandlerPlugin.onError` catches DomainError and maps to uniform body. Revised resolver:

```typescript
import { UnauthenticatedError } from '../../domain/errors'

// ...in the .macro resolve body:
async resolve({ request: { headers } }) {
  const rawApiKey = headers.get('x-api-key')
  if (rawApiKey !== null) {
    const ctx = await identity.verifyApiKey(rawApiKey)
    if (!ctx) throw new UnauthenticatedError('Authentication required')
    return { authContext: ctx satisfies AuthContext }
  }
  const sessionCtx = await identity.verifySession(headers)
  if (!sessionCtx) throw new UnauthenticatedError('Authentication required')
  return { authContext: sessionCtx satisfies AuthContext }
}
```

If Elysia macro resolvers swallow thrown errors and return 500 instead (breaking D-12 contract), fall back to Option B: pass `set` into the resolve destructure and include `requestId = (set.headers['x-request-id'] as string) ?? 'unknown'` in a manual body, while still returning `status(401, body)`. Verify via `tests/integration/auth/401-body-shape.test.ts`: if body has `requestId` matching response header, pick the path that works.

  test('1000-iteration latency: |t_malformed - t_valid_wrong_hash| / t_valid_wrong_hash < 0.2', async () => {

    const malformedKey = 'not-rig-format'
    // Valid prefix but mutated suffix → triggers real findByPrefix miss (no row)

      await app.handle(new Request('http://localhost/me', { headers: { 'x-api-key': validFormatWrongKey } }))
    }

    let malTotal = 0
    let wrongHashTotal = 0
    for (let i = 0; i < N; i++) {

      wrongHashTotal += performance.now() - t2
    }
    const malMean = malTotal / N
    const wrongHashMean = wrongHashTotal / N
    const delta = Math.abs(malMean - wrongHashMean)
    const ratio = delta / wrongHashMean
    console.log(JSON.stringify({ test: 'timing-safe-apikey', malMean, wrongHashMean, ratio }))
