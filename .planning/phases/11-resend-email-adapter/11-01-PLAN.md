```typescript
import { Resend } from 'resend'
import type { Logger } from 'pino'
import type { IEmailPort } from '../../application/ports/email.port'

export class ResendEmailAdapter implements IEmailPort {
  private readonly client: Resend

  constructor(
    apiKey: string,
    private readonly from: string,
    private readonly logger: Logger,
  ) {
    this.client = new Resend(apiKey)
  }

  async send(params: { to: string; subject: string; body: string }): Promise<void> {
    const { data, error } = await this.client.emails.send({
      from: this.from,
      to: params.to,
      subject: params.subject,
      text: params.body,
    })

    if (error) {
      this.logger.error({ to: params.to, subject: params.subject, error }, 'ResendEmailAdapter: send failed')
      throw new Error(`Failed to send email via Resend: ${error.message}`)
    }

    this.logger.info({ to: params.to, subject: params.subject, id: data?.id }, 'ResendEmailAdapter: email sent')
  }
}
```

Notes:
- Constructor signature: (apiKey: string, from: string, logger: Logger) — per D-04
- Builds new Resend(apiKey) in constructor — per D-04
- Calls this.client.emails.send({...}) — per D-04
- Uses text: params.body (plain text only) — per D-05
- from: this.from (from env var passed as constructor arg) — per D-05
- On error: logger.error + throw Error — per D-04
- On success: logger.info at info level (agent's discretion)

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| env → ConfigSchema | RESEND_API_KEY and RESEND_FROM_ADDRESS enter the process via environment; TypeBox validates at startup |
| createAuthModule → Resend SDK | apiKey is passed to Resend SDK constructor; any leak would expose the secret |
| ResendEmailAdapter → Resend API | email body (a URL string from BetterAuth) is sent over HTTPS to Resend's API |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-11-01 | Information Disclosure | RESEND_API_KEY in config | mitigate | Key is read once at startup into Config, never logged. ResendEmailAdapter receives it as constructor arg and stores in private Resend client — never exposed on logger calls. |
| T-11-02 | Spoofing | from: address in email send | accept | from is set by the operator via RESEND_FROM_ADDRESS env var. Resend enforces domain verification on their side. No user input controls the from field. |
| T-11-03 | Denial of Service | Resend API errors causing request failure | mitigate | Errors are caught, logged, and re-thrown as Error objects. BetterAuth's own error handling propagates the failure to the caller without crashing the process. |
| T-11-04 | Elevation of Privilege | Email body (URL) injected by BetterAuth | accept | Body is a URL string generated internally by BetterAuth (not user-controlled). sent as text: field (plain text), not HTML — XSS is impossible in plain-text email. |
| T-11-05 | Repudiation | Email delivery confirmation | accept | Resend dashboard provides delivery logs. Adapter logs the email id on success at info level. Sufficient for v1.3. |
</threat_model>
